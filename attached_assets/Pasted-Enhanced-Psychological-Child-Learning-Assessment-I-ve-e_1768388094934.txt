        // Enhanced Assessment Logic
        const sections = document.querySelectorAll('.step-section');
        const progressBar = document.getElementById('progress-bar');
        const stepLabel = document.getElementById('step-label');
        const stepCount = document.getElementById('step-count');
        const prevBtn = document.getElementById('prevBtn');
        const nextBtn = document.getElementById('nextBtn');
        const navArea = document.getElementById('nav-area');
        
        let currentStep = 0;
        const totalSteps = sections.length;
        
        const stepTitles = [
            "Student Profile", "Learning Challenges", "Attention & Focus",
            "Emotional Drivers", "Learning Behavior", "Motivation Patterns",
            "Learning Style", "Communication", "Final Details"
        ];

        // Initialize
        updateUI();

        // Event Listeners
        nextBtn.addEventListener('click', () => {
            if (validateStep(currentStep)) {
                if (currentStep < totalSteps - 1) {
                    currentStep++;
                    animateStepTransition();
                } else {
                    submitAssessment();
                }
            } else {
                showValidationError();
            }
        });

        prevBtn.addEventListener('click', () => {
            if (currentStep > 0) {
                currentStep--;
                animateStepTransition();
            }
        });

        // Functions
        function animateStepTransition() {
            const currentSection = sections[currentStep];
            const cardContent = document.querySelector('.card-content');
            
            // Fade out current
            cardContent.style.opacity = '0';
            cardContent.style.transform = 'translateX(-20px)';
            
            setTimeout(() => {
                updateUI();
                cardContent.style.opacity = '1';
                cardContent.style.transform = 'translateX(0)';
            }, 300);
        }

        function updateUI() {
            // Update active section
            sections.forEach((sec, idx) => {
                sec.classList.remove('active');
                if (idx === currentStep) {
                    sec.classList.add('active');
                }
            });

            // Update progress
            const progress = ((currentStep + 1) / totalSteps) * 100;
            progressBar.style.width = `${progress}%`;
            stepLabel.textContent = stepTitles[currentStep];
            stepCount.textContent = `Step ${currentStep + 1} of ${totalSteps}`;

            // Update buttons
            prevBtn.disabled = currentStep === 0;
            prevBtn.style.opacity = currentStep === 0 ? '0' : '1';
            
            if (currentStep === totalSteps - 1) {
                nextBtn.innerHTML = 'Generate Learning Profile ‚ú®';
            } else {
                nextBtn.innerHTML = 'Next ‚Üí';
            }

            // Scroll to top smoothly
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function validateStep(stepIndex) {
            const currentSection = sections[stepIndex];
            const requiredInputs = currentSection.querySelectorAll('input[required], input[type="radio"]');
            const radioGroups = new Set();
            let allValid = true;

            // Check radio groups
            requiredInputs.forEach(input => {
                if (input.type === 'radio') {
                    radioGroups.add(input.name);
                }
            });

            // Validate each radio group has a selection
            for (let groupName of radioGroups) {
                const checked = currentSection.querySelector(`input[name="${groupName}"]:checked`);
                if (!checked) {
                    allValid = false;
                    highlightUnanswered(groupName);
                }
            }

            return allValid;
        }

        function highlightUnanswered(groupName) {
            const inputs = document.querySelectorAll(`input[name="${groupName}"]`);
            inputs.forEach(input => {
                const card = input.nextElementSibling;
                card.style.borderColor = '#EF4444';
                card.style.animation = 'shake 0.5s ease';
                
                setTimeout(() => {
                    card.style.borderColor = '';
                    card.style.animation = '';
                }, 500);
            });
        }

        function showValidationError() {
            const formCard = document.querySelector('.form-card');
            formCard.style.animation = 'none';
            setTimeout(() => {
                formCard.style.animation = 'shake 0.5s ease';
            }, 10);
        }

        function submitAssessment() {
            // Hide navigation and show analysis
            navArea.style.display = 'none';
            document.querySelector('.progress-wrapper').style.opacity = '0.5';
            sections[currentStep].classList.remove('active');
            
            const overlay = document.getElementById('analysis-overlay');
            const text = document.getElementById('analysis-text');
            overlay.style.display = 'flex';

            // Analysis messages with psychological insights
            const analysisSteps = [
                "Analyzing cognitive patterns...",
                "Assessing emotional learning drivers...",
                "Identifying focus & attention style...",
                "Mapping motivation triggers...",
                "Calculating optimal learning pace...",
                "Matching with specialized tutors..."
            ];

            let step = 0;
            const analysisInterval = setInterval(() => {
                if (step < analysisSteps.length) {
                    text.textContent = analysisSteps[step];
                    step++;
                } else {
                    clearInterval(analysisInterval);
                    showResults();
                }
            }, 1200);
        }

        function showResults() {
            document.getElementById('analysis-overlay').style.display = 'none';
            
            // Determine result based on inputs (simplified logic)
            const styleInput = document.querySelector('input[name="learning_style"]:checked');
            const focusInput = document.querySelector('input[name="focus"]:checked');
            const motivationInput = document.querySelector('input[name="motivation"]:checked');
            
            let tutorType = "Expert Tutor";
            
            if (styleInput && focusInput) {
                if (styleInput.value === 'visual' && focusInput.value === 'short') {
                    tutorType = "Visual & Interactive Specialist";
                } else if (motivationInput && motivationInput.value === 'internal') {
                    tutorType = "Inquiry-Based Learning Expert";
                } else if (motivationInput && motivationInput.value === 'praise') {
                    tutorType = "Encouragement-Focused Mentor";
                }
            }
            
            document.getElementById('dynamic-result').textContent = tutorType;
            document.getElementById('result-screen').style.display = 'block';
        }

        const style = document.createElement('style');

            }
