        let currentStep = 0;
        const totalSteps = 9; // Contact + 8 assessment steps
        
        const stepTitles = [

            "Motivation Patterns", "Communication Style", "Learning Goals"
        ];

        nextBtn.addEventListener('click', () => {
            if (validateStep(currentStep)) {
                if (currentStep < totalSteps - 1) {
                    currentStep++;
                    animateStepTransition();
                } else {
                    // Last step - show loading and generate report
                    showLoadingScreen();
                }
            } else {
                showValidationError();
            }
        });

        prevBtn.addEventListener('click', () => {
            if (currentStep > 0) {
                currentStep--;
                animateStepTransition();
            }
        });

        // Download Report Button
        document.getElementById('downloadReportBtn')?.addEventListener('click', generatePDFReport);
        
        // Verify & Submit Button
        document.getElementById('verifySubmitBtn')?.addEventListener('click', submitAssessment);

        // Input validation for contact fields
        document.getElementById('studentName')?.addEventListener('blur', validateName);
        document.getElementById('parentName')?.addEventListener('blur', validateName);
        document.getElementById('email')?.addEventListener('blur', validateEmail);
        document.getElementById('mobile')?.addEventListener('blur', validateMobile);

        // Functions
        function animateStepTransition() {
            const cardContent = document.querySelector('.card-content');
            
            // Fade out current
            cardContent.style.opacity = '0';
            cardContent.style.transform = 'translateX(-20px)';
            
            setTimeout(() => {
                updateUI();
                cardContent.style.opacity = '1';
                cardContent.style.transform = 'translateX(0)';
            }, 300);
        }

        function updateUI() {
            // Update active section
            sections.forEach((sec, idx) => {
                sec.classList.remove('active');
                if (idx === currentStep) {
                    sec.classList.add('active');
                }
            });

            // Update progress
            const progress = ((currentStep + 1) / totalSteps) * 100;
            progressBar.style.width = `${progress}%`;
            stepLabel.textContent = stepTitles[currentStep];
            stepCount.textContent = `Step ${currentStep + 1} of ${totalSteps}`;

            // Update buttons
            prevBtn.disabled = currentStep === 0;
            prevBtn.style.opacity = currentStep === 0 ? '0' : '1';
            
            if (currentStep === totalSteps - 1) {
                nextBtn.innerHTML = 'Complete Assessment ✨';
            } else {
                nextBtn.innerHTML = 'Next →';
            }

            // Scroll to top smoothly
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function validateStep(stepIndex) {
            // Special validation for contact step
            if (stepIndex === 0) {
                return validateContactStep();
            }
            
            const currentSection = sections[stepIndex];
            const requiredInputs = currentSection.querySelectorAll('input[required], input[type="radio"]');
            const radioGroups = new Set();
            let allValid = true;

            // Check radio groups
            requiredInputs.forEach(input => {
                if (input.type === 'radio') {
                    radioGroups.add(input.name);
                }
            });

            // Validate each radio group has a selection
            for (let groupName of radioGroups) {
                const checked = currentSection.querySelector(`input[name="${groupName}"]:checked`);
                if (!checked) {
                    allValid = false;
                    highlightUnanswered(groupName);
                }
            }

            return allValid;
        }

        function validateContactStep() {
            let isValid = true;
            
            // Validate name
            if (!validateNameField('studentName')) {
                isValid = false;
                showError('studentName', 'Please enter student name');
            }
            
            if (!validateNameField('parentName')) {
                isValid = false;
                showError('parentName', 'Please enter parent name');
            }
            
            // Validate email
            if (!validateEmailField()) {
                isValid = false;
                showError('email', 'Please enter a valid email address');
            }
            
            // Validate mobile
            if (!validateMobileField()) {
                isValid = false;
                showError('mobile', 'Please enter a valid mobile number');
            }
            
            return isValid;
        }

        function validateNameField(fieldId) {
            const field = document.getElementById(fieldId);
            const value = field.value.trim();
            return value.length >= 2;
        }

        function validateEmailField() {
            const email = document.getElementById('email').value.trim();
            const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            return emailRegex.test(email);
        }

        function validateMobileField() {
            const mobile = document.getElementById('mobile').value.trim();
            const mobileRegex = /^[\+]?[0-9\s\-\(\)]{10,}$/;
            return mobileRegex.test(mobile);
        }

        function validateName() {
            const fieldId = this.id;
            const isValid = validateNameField(fieldId);
            if (isValid) {
                hideError(fieldId);
            } else {
                showError(fieldId, 'Please enter a valid name');
            }
        }

        function validateEmail() {
            const isValid = validateEmailField();
            if (isValid) {
                hideError('email');
            } else {
                showError('email', 'Please enter a valid email address');
            }
        }

        function validateMobile() {
            const isValid = validateMobileField();
            if (isValid) {
                hideError('mobile');
            } else {
                showError('mobile', 'Please enter a valid mobile number');
            }
        }

        function showError(fieldId, message) {
            const field = document.getElementById(fieldId);
            const errorElement = document.getElementById(fieldId + 'Error');
            
            field.classList.add('error');
            errorElement.textContent = message;
            errorElement.style.display = 'block';
        }

        function hideError(fieldId) {
            const field = document.getElementById(fieldId);
            const errorElement = document.getElementById(fieldId + 'Error');
            
            field.classList.remove('error');
            errorElement.style.display = 'none';
        }

        function highlightUnanswered(groupName) {
            const inputs = document.querySelectorAll(`input[name="${groupName}"]`);
            inputs.forEach(input => {
                const card = input.nextElementSibling;
                card.style.borderColor = '#EF4444';
                card.style.animation = 'shake 0.5s ease';
                
                setTimeout(() => {
                    card.style.borderColor = '';
                    card.style.animation = '';
                }, 500);
            });
        }

        function showValidationError() {

            setTimeout(() => {
                formCard.style.animation = 'shake 0.5s ease';

        function showLoadingScreen() {

            collectAssessmentData();
            
            // Hide navigation and show loading
            navArea.style.display = 'none';
            sections[currentStep].classList.remove('active');
            loadingScreen.style.display = 'flex';
            
            // Update loading messages
            const loadingText = document.getElementById('loading-text');
            const loadingSubtext = document.getElementById('loading-subtext');
            
            const messages = [
                { text: "Analyzing cognitive patterns...", subtext: "Processing learning style preferences" },
                { text: "Assessing emotional drivers...", subtext: "Evaluating motivation and resilience factors" },
                { text: "Calculating attention metrics...", subtext: "Determining optimal session structure" },
                { text: "Generating personalized insights...", subtext: "Creating customized learning strategies" },

            let i = 0;
            const interval = setInterval(() => {
                if (i < messages.length) {
                    loadingText.textContent = messages[i].text;
                    loadingSubtext.textContent = messages[i].subtext;
                    i++;
                } else {
                    clearInterval(interval);
                    setTimeout(showReportScreen, 1500);
                }
            }, 1500);

        function collectAssessmentData() {
